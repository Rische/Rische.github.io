var radius = 320 / pi
var tex = surface_get_texture(input)

surface_set_target(output)
draw_clear_alpha(c_black, 0)

if shadow
  {
  shader_set(shd_backdrop)
  draw_primitive_begin_texture(pr_trianglestrip, tex)
  for(var i=0; i<=1; i+=.05)
   {
   var xx = 320 / 2 + cos(pi+i * pi) * radius
   draw_vertex_texture(xx+2, 0+2, i, 0)
   draw_vertex_texture(xx+2, 240+2, i, 1)
   }
  draw_primitive_end()
  shader_reset()
  }

shader_set(shd_outside_light)
shader_set_uniform_f(global.SP_darkness, .5)
for(var j=dep; j>0; j--)
  {
  draw_primitive_begin_texture(pr_trianglestrip, tex)
  for(var i=0; i<=1; i+=.05)
   {
   var xx = 320 / 2 + cos(pi+i*pi) * (radius-j)
   draw_vertex_texture_colour(xx, 0, i, 0, c_white, 1)
   draw_vertex_texture_colour(xx, 240, i, 1, c_white, 1)
   }
  draw_primitive_end()
  }

shader_set_uniform_f(global.SP_darkness, 0)

draw_primitive_begin_texture(pr_trianglestrip, tex)
for(var i=0; i<=1; i+=.05)
  {

  draw_vertex_texture(xx, 0, i, 0)
  draw_vertex_texture(xx, 240, i, 1)
  }
